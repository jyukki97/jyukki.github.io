while (!p.empty()) {
	w = p.top().second;
	p.pop();
	if (b[w])	continue;
	b[w] = true;
	for (auto t : a[w]) {
		u = t.first, v = t.second;
	if (d[u] > d[w] + v) {
		d[u] = d[w] + v;
		p.push({ d[u], u });
	}
}
